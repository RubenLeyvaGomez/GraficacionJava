  FIN METODO set_cota

  INICIO

  FIN METODO set_angulo

  FIN METODO get_angulo

  METODO PUBLICO 2d_to_3d
  SECCION DE VARIABLES
    m1: REAL 
    n1: REAL
    m: ENTERO
    n: ENTERO
    abs: ENTERO
    ord: ENTERO
  INICIO 
    m1 <-- cota*cos(angulo)
    n1 <-- cota*sen(angulo)
    m <-- redondear(m1)
    n <-- redondear(n1)
    
    mipunto2d OBJETO DE CL_PUNTO

    abs <-- m + abscisa
    ord <-- n + ordenada
    
    mipunto2d.set_abscisa(abs)
    mipunto2d.set_ordenada(ord)

    REGRESAR mipunto2d

  FIN METODO 2d_to_3d

  METODO PUBLICO CL_PUNTO3D(abs, ord, cota: ENTERO)
  INICIO 
  abscisa <-- abs
  ordenada <-- ord
  cota <-- cota

  FIN METODO CL_PUNTO3D


FIN CLASE CL_PUNTO3D
