METODO CL_ESPACIO_GEOMETRICO(o: CL_PUNTO, ex: CL_LINEA, ey: CL_LINEA,ez: CL_LINEA, eg: CL_ESPACIO_GRAFICO, pz: CL_PUNTO, x1: ENTERO, x2: ENTERO)
INICIO 
  origen <-- o
  ejex <-- ex
  ejey <-- ey
  ejez <-- ez
  n:ENTERO
  m: ENTERO
  n1: REAL
  m1: REAL 
  d: ENTERO

  d <-- eg.get_ancho/2
  n1 <-- d*sin(get_angulo*3.1416)
  m1 <-- d*cos(get_angulo*3.1416)  
  n <-- Redondear(n1)
  m <-- Redondear(m1)
  
  pz.set_abscisa(n)
  pz.set_ordenada(m)
  pz.geo_to_graf(x1,x2)
  
  ejex.set_p_inicial.set_abscisa(0)
  ejex.set_p_inicial.set_ordenada(origen.get_ordenada)
  eg.get_ancho
  ejex.set_p_final.set_abscisa(eg.get_ancho)
  ejex.set_p_final.set_ordenada(origen.get_ordenada)

  ejey.set_p_inicial.set_abscisa(origen.get_abscisa)
  ejey.set_p_inicial.set_ordenada(0)

  ejey.set_p_final.set_abscisa(origen.get_abscisa)
  ejey.set_p:final.set_ordenada(eg.get_alto)

  ejez.set_p_inicial.set_abscisa(origen.get_abscisa)
  ejez.set_p_inicial.set_ordenada(origen.get_ordenada)

  ejez.set_p_final.set_abscisa(pz.get_abscisa)
  ejez.set_p_final.set_abscisa(pz.get_ordenada)

FIN METODO CL_ESPACIO_GEOMETRICO
